## Setup Guide

### 1. Create tRPC Instance

```typescript title="src/trpc.ts"
import { initTRPC } from '@trpc/server';
import { ZodError } from 'zod';

const t = initTRPC.create({
  errorFormatter({ shape, error }) {
    return {
      ...shape,
      data: {
        ...shape.data,
        zodError:
          error.cause instanceof ZodError ? error.cause.flatten() : null,
      },
    };
  },
});

export const router = t.router;
export const publicProcedure = t.procedure;
```

### 2. Create Your API Router

```typescript title="src/router.ts"
import { z } from 'zod';
import { publicProcedure, router } from './trpc';

// Mock database
const posts: Array<{
  id: string;
  title: string;
  content: string;
  createdAt: Date;
}> = [];

export const appRouter = router({
  hello: publicProcedure
    .input(z.object({ name: z.string().optional() }))
    .query(({ input }) => {
      return {
        greeting: `Hello ${input.name ?? 'World'}!`,
      };
    }),

  getPosts: publicProcedure.query(() => {
    return posts;
  }),

  createPost: publicProcedure
    .input(
      z.object({
        title: z.string().min(1),
        content: z.string().min(1),
      }),
    )
    .mutation(({ input }) => {
      const post = {
        id: Date.now().toString(),
        ...input,
        createdAt: new Date(),
      };
      posts.push(post);
      return post;
    }),

  deletePost: publicProcedure
    .input(z.object({ id: z.string() }))
    .mutation(({ input }) => {
      const index = posts.findIndex((p) => p.id === input.id);
      if (index === -1) {
        throw new Error('Post not found');
      }
      const deleted = posts.splice(index, 1)[0];
      return deleted;
    }),
});

export type AppRouter = typeof appRouter;
```

### 3. Setup Express Server

```typescript title="src/server.ts"
import { createExpressMiddleware } from '@trpc/server/adapters/express';
import express from 'express';
import { appRouter } from './router';

const app = express();
const port = process.env.PORT || 3000;

// Create context
const createContext = ({
  req,
  res,
}: {
  req: express.Request;
  res: express.Response;
}) => {
  return {
    req,
    res,
    // Add user info, database connections, etc.
  };
};

// Add tRPC middleware
app.use(
  '/api/trpc',
  createExpressMiddleware({
    router: appRouter,
    createContext,
  }),
);

// Health check endpoint
app.get('/health', (req, res) => {
  res.json({ status: 'ok', timestamp: new Date().toISOString() });
});

app.listen(port, () => {
  console.log(`ðŸš€ Server running at http://localhost:${port}`);
  console.log(`ðŸ“¡ tRPC endpoint at http://localhost:${port}/api/trpc`);
});
```

## Add Authentication

Here's how to add JWT authentication:

```typescript title="src/auth.ts"
import jwt from 'jsonwebtoken';

const JWT_SECRET = process.env.JWT_SECRET || 'your-secret-key';

export function verifyToken(token: string) {
  try {
    return jwt.verify(token, JWT_SECRET) as { userId: string; email: string };
  } catch {
    return null;
  }
}

export function signToken(payload: { userId: string; email: string }) {
  return jwt.sign(payload, JWT_SECRET, { expiresIn: '7d' });
}
```

```typescript title="src/context.ts"
import express from 'express';
import { verifyToken } from './auth';

export const createContext = ({
  req,
  res,
}: {
  req: express.Request;
  res: express.Response;
}) => {
  // Extract token from Authorization header
  const token = req.headers.authorization?.replace('Bearer ', '');
  const user = token ? verifyToken(token) : null;

  return {
    req,
    res,
    user,
  };
};

export type Context = ReturnType<typeof createContext>;
```

```typescript title="src/trpc.ts"
import { initTRPC, TRPCError } from '@trpc/server';
import type { Context } from './context';

const t = initTRPC.context<Context>().create();

export const router = t.router;
export const publicProcedure = t.procedure;

// Protected procedure
export const protectedProcedure = t.procedure.use(({ ctx, next }) => {
  if (!ctx.user) {
    throw new TRPCError({ code: 'UNAUTHORIZED' });
  }
  return next({
    ctx: {
      user: ctx.user,
    },
  });
});
```

```typescript title="src/router.ts"
export const appRouter = router({
  // Public routes
  hello: publicProcedure.query(() => ({ message: 'Hello World!' })),

  // Protected routes
  getProfile: protectedProcedure.query(({ ctx }) => {
    return {
      userId: ctx.user.userId,
      email: ctx.user.email,
    };
  }),

  createPost: protectedProcedure
    .input(
      z.object({
        title: z.string().min(1),
        content: z.string().min(1),
      }),
    )
    .mutation(async ({ input, ctx }) => {
      // User is guaranteed to exist here
      return await createPost({
        ...input,
        authorId: ctx.user.userId,
      });
    }),
});
```

## Client Usage

Create a client to consume your API:

```typescript title="client.ts"
import { createTRPCClient, httpBatchLink } from '@trpc/client';
import type { AppRouter } from './src/router';

const client = createTRPCClient<AppRouter>({
  links: [
    httpBatchLink({
      url: 'http://localhost:3000/api/trpc',
      headers() {
        const token = localStorage.getItem('auth-token');
        return token ? { authorization: `Bearer ${token}` } : {};
      },
    }),
  ],
});

// Usage examples
async function examples() {
  // Query
  const hello = await client.hello.query({ name: 'tRPC' });
  console.log(hello.greeting);

  // Mutation
  const post = await client.createPost.mutate({
    title: 'My Post',
    content: 'Hello World!',
  });
  console.log('Created post:', post);

  // Get all posts
  const posts = await client.getPosts.query();
  console.log('All posts:', posts);
}
```

## Common Issues

<Accordions>
<Accordion title="CORS Issues">
If your frontend can't connect to your API, make sure CORS is properly configured:

```typescript
app.use(
  cors({
    origin: ['http://localhost:3000', 'https://yourapp.com'],
    credentials: true,
  }),
);
```

</Accordion>

<Accordion title="Type Errors">
Make sure your TypeScript configuration is correct and all dependencies are installed:

```bash
npm install -D typescript @types/node @types/express
```

</Accordion>

<Accordion title="Port Already in Use">
If port 3000 is already in use, change it in your server.ts file or use an environment variable:

```typescript
const port = process.env.PORT || 3001;
```

</Accordion>
</Accordions>
