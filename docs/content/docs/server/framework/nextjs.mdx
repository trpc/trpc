## Setup Guide

### 1. Create tRPC Instance

```typescript title="src/server/trpc.ts"
import { initTRPC } from '@trpc/server';
import { ZodError } from 'zod';

const t = initTRPC.create({
  errorFormatter({ shape, error }) {
    return {
      ...shape,
      data: {
        ...shape.data,
        zodError:
          error.cause instanceof ZodError ? error.cause.flatten() : null,
      },
    };
  },
});

export const router = t.router;
export const publicProcedure = t.procedure;
```

### 2. Create Your API Router

```typescript title="src/server/router.ts"
import { z } from 'zod';
import { publicProcedure, router } from './trpc';

export const appRouter = router({
  hello: publicProcedure
    .input(z.object({ name: z.string().optional() }))
    .query(({ input }) => {
      return {
        greeting: `Hello ${input.name ?? 'World'}!`,
      };
    }),

  createPost: publicProcedure
    .input(
      z.object({
        title: z.string().min(1),
        content: z.string().min(1),
      }),
    )
    .mutation(({ input }) => {
      // Here you would save to your database
      return {
        id: Date.now().toString(),
        ...input,
        createdAt: new Date(),
      };
    }),
});

export type AppRouter = typeof appRouter;
```

### 4. Setup tRPC Client

```typescript title="src/utils/trpc.ts"
import { createTRPCClient, httpBatchLink } from '@trpc/client';
import type { AppRouter } from '../server/router';

export const trpc = createTRPCClient<AppRouter>({
  links: [
    httpBatchLink({
      url: '/api/trpc',
    }),
  ],
});
```

<Tab value="Client Component">
```typescript title="src/components/CreatePost.tsx"
'use client';

import { useState } from 'react';
import { trpc } from '../utils/trpc';

export default function CreatePost() {
  const [title, setTitle] = useState('');
  const [content, setContent] = useState('');

const handleSubmit = async (e: React.FormEvent) => {
e.preventDefault();

    try {
      const post = await trpc.createPost.mutate({ title, content });
      console.log('Created post:', post);

      // Reset form
      setTitle('');
      setContent('');
    } catch (error) {
      console.error('Error creating post:', error);
    }

};

return (

<form onSubmit={handleSubmit}>
  <input
    type="text"
    value={title}
    onChange={(e) => setTitle(e.target.value)}
    placeholder="Post title"
    required
  />
  <textarea
    value={content}
    onChange={(e) => setContent(e.target.value)}
    placeholder="Post content"
    required
  />
  <button type="submit">Create Post</button>
</form>
); }

````

### Setup Providers

<Tabs items={['App Router', 'Pages Router']}>
<Tab value="App Router">
```typescript title="src/app/providers.tsx"
'use client';

import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { httpBatchLink } from '@trpc/client';
import { useState } from 'react';
import { trpc } from '../utils/trpc';

export default function Providers({ children }: { children: React.ReactNode }) {
  const [queryClient] = useState(() => new QueryClient());
  const [trpcClient] = useState(() =>
    trpc.createClient({
      links: [
        httpBatchLink({
          url: '/api/trpc',
        }),
      ],
    }),
  );

return (

<trpc.Provider client={trpcClient} queryClient={queryClient}>
  <QueryClientProvider client={queryClient}>{children}</QueryClientProvider>
</trpc.Provider>
); }

````

```typescript title="src/app/layout.tsx"
import Providers from './providers';

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en">
      <body>
        <Providers>
          {children}
        </Providers>
      </body>
    </html>
  );
}
````

```typescript title="src/pages/_app.tsx"
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { httpBatchLink } from '@trpc/client';
import { useState } from 'react';
import { trpc } from '../utils/trpc';
import type { AppType } from 'next/app';

const MyApp: AppType = ({ Component, pageProps }) => {
const [queryClient] = useState(() => new QueryClient());
const [trpcClient] = useState(() =>
trpc.createClient({
links: [
httpBatchLink({
url: '/api/trpc',
}),
],
}),
);

return (

<trpc.Provider client={trpcClient} queryClient={queryClient}>
  <QueryClientProvider client={queryClient}>
    <Component {...pageProps} />
  </QueryClientProvider>
</trpc.Provider>
); };

export default trpc.withTRPC(MyApp);
```

### Use React Query Hooks

```typescript title="src/components/PostList.tsx"
'use client';

import { trpc } from '../utils/trpc';

export default function PostList() {
  const { data: hello, isLoading } = trpc.hello.useQuery({ name: 'tRPC' });
  const createPost = trpc.createPost.useMutation();

  if (isLoading) return <div>Loading...</div>;

  return (
    <div>
      <h1>{hello?.greeting}</h1>
      <button
        onClick={() =>
          createPost.mutate({
            title: 'My Post',
            content: 'Hello World!'
          })
        }
        disabled={createPost.isLoading}
      >
        {createPost.isLoading ? 'Creating...' : 'Create Post'}
      </button>
    </div>
  );
}
```

## Common Issues

<Accordions>
<Accordion title="Module not found errors">
Make sure you've installed all required dependencies and your TypeScript configuration is correct.

```json title="tsconfig.json"
{
  "compilerOptions": {
    "strict": true,
    "moduleResolution": "node",
    "esModuleInterop": true,
    "allowSyntheticDefaultImports": true
  }
}
```

</Accordion>

<Accordion title="CORS issues">
If you're calling your API from a different domain, you may need to configure CORS:

```typescript title="src/pages/api/trpc/[trpc].ts"
import { createNextApiHandler } from '@trpc/server/adapters/next';
import { appRouter } from '../../../server/router';

export default createNextApiHandler({
  router: appRouter,
  createContext: () => ({}),
  responseMeta() {
    return {
      headers: {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET,OPTIONS,PATCH,DELETE,POST,PUT',
        'Access-Control-Allow-Headers': 'Content-Type, Authorization',
      },
    };
  },
});
```

</Accordion>
</Accordions>
